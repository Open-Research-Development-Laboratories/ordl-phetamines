Agent Dispatch Example

dispatch_agent.py 
import ordl

# Initialize the ORDL client with your API key
client = ordl.Client("your_api_key_here")

# Create a new agent with specific capabilities
agent = client.agents.create(
    name="data-processor",
    model="gpt-4-turbo",
    capabilities=["text-analysis", "data-extraction"],
    fleet="production-us-east"
)

# Dispatch the agent with a task
task = agent.dispatch(
    instruction="Analyze customer feedback and extract key themes",
    input_data=feedback_dataset,
    priority="high",
    timeout=300
)

# Monitor task progress
while not task.complete:
    print(f"Progress: {task.progress}%")
    task.refresh()

print(f"Results: {task.results}")
